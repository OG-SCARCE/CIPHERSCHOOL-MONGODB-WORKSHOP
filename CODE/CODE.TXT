2. Inserting Documents

// Insert a single document.
// BEST PRACTICE: Standardized field names to 'task' (singular), 'createdAt', and 'isCompleted' (camelCase).
db.tasks.insertOne({
  task: "Learning MongoDB",
  createdAt: "27th Feb, 2026",
  deadline: "28th Feb, 2026",
  user: "Raj",
  isCompleted: false
});

3. Reading and Updating Documents

// Retrieve all documents in the 'tasks' collection
db.tasks.find();

// Retrieve all tasks assigned specifically to the user "SKT"
db.tasks.find({ user: "SKT" });

// Update the FIRST document that matches the filter { user: "SKT" }.
// The $set operator modifies an existing field or adds a new one if it doesn't exist.
db.tasks.updateOne(
  { user: "SKT" },
  { $set: { deadline: "29th Feb, 2026" } }
);

// Update ALL documents that match the filter { user: "Raj" }.
db.tasks.updateMany(
  { user: "Raj" },
  { $set: { deadline: "29th Feb, 2026" } }
);

// Without this, your $gt (greater than) queries at the end of the script would return nothing.
db.tasks.updateMany(
  {}, // An empty filter matches every document in the collection
  { $set: { procrastinated: 0 } }
);

// Update multiple documents using the $or logical operator.
db.tasks.updateMany(
  { 
    $or: [
      { task: "workout", user: "Raj" },
      { task: "Teaching", user: "SKT" }
    ] 
  }, 
  { $set: { isCompleted: true } } 
);


------------------------------------------------------------------------

4. Deleting & Advanced Querying

// Delete the FIRST document that matches this exact task name
db.tasks.deleteOne({ task: "Learning MongoDB" });

// Delete ALL documents where the 'isCompleted' status is true
db.tasks.deleteMany({ isCompleted: true });

// Retrieve tasks where the 'procrastinated' count is Greater Than ($gt) 0

db.tasks.find({ procrastinated: { $gt: 0 } });

// Retrieve tasks where the 'procrastinated' count is Greater Than ($gt) 3
db.tasks.find({ procrastinated: { $gt: 3 } });

// Alternative syntax to access a collection.
// This bracket notation is highly useful if your collection name contains spaces or hyphens (e.g., db["my tasks"]).
db["tasks"].find();
